\section{Research activity}

\subsection{$CP$ violation measurements with $B^0$ and $B^0_s$ decays}
\cvitem{}{I performed measurements of the CKM angle $\gamma$ with time-dependent analyses of $B^0\to D^{\mp}\pi^{\pm}$ and $B_s^0\to D_s^{\mp}K^{\pm}$ using
data collected with the LHCb detector. The result obtained with $B^0\to D^{\mp}\pi^{\pm}$ is more precise than previous measurements from BaBar and Belle collaborations.
The analysis of $B_s^0\to D_s^{\mp}K^{\pm}$ is the first measurement of $\gamma$ using $B^0_s$ decays.}

\subsection{Flavour tagging algorithms}
\cvitem{}{I implemented and optimised flavour tagging algorithms used by the LHCb collaboration to infer the flavour of neutral $B$ mesons for time-dependent analyses.}

\subsection{Silicon Tracker}
\cvitem{}{I was responsible for the calibration and the maintenance of the Silicon Tracker detector of LHCb.}
